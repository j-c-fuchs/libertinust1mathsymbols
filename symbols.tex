\section{Math alphabets}

The following math alphabets are available.
Note that the option \verb|lcgreekalpha| is needed to make the lowercase
greek letters follow the math alphabet.
A table of all the greek letters is given in \cref{tab:greek-letters}.
Note that for all greek letters, there exist upright and italic
commands, too: \verb|\upalpha| and \verb|\alphaup| for $\upalpha$,
\verb|\italpha| and \verb|\alphait| for $\italpha$ etc.

\begin{description}
\item[default] same as italic
\item[roman] \verb|\mathrm| \\
  $\mathrm{\abc}$ $\mathrm{\ABC}$ $\mathrm{\nums}$ \\
  $\mathrm{\greekabc}$ $\mathrm{\greekABC}$
\item[bold] \verb|\mathbf| \\
  $\mathbf{\abc}$ $\mathbf{\ABC}$ $\mathbf{\nums}$ \\
  $\mathbf{\greekabc}$ $\mathbf{\greekABC}$
\item[italic] \verb|\mathit| \\
  $\mathit{\abc}$ $\mathit{\ABC}$ $\mathit{\nums}$ \\
  $\mathit{\greekabc}$ $\mathit{\greekABC}$
\item[bold italic] \verb|\mathbfit| \\
  $\mathbfit{\abc}$ $\mathbfit{\ABC}$ $\mathbfit{\nums}$ \\
  $\mathbfit{\greekabc}$ $\mathbfit{\greekABC}$
\item[sans] \verb|\mathsf| \\
  $\mathsf{\abc}$ $\mathsf{\ABC}$ $\mathsf{\nums}$ \\
  $\mathsf{\greekabc}$ $\mathsf{\greekABC}$
\item[bold sans] \verb|\mathsfbf| \\
  $\mathsfbf{\abc}$ $\mathsfbf{\ABC}$ $\mathsfbf{\nums}$ \\
  $\mathsfbf{\greekabc}$ $\mathsfbf{\greekABC}$
\item[italic sans] \verb|\mathsfit| (needs the \verb|mathsfit| option) \\
  $\mathsfit{\abc}$ $\mathsfit{\ABC}$ $\mathsfit{\nums}$ \\
  $\mathsfit{\greekabc}$ $\mathsfit{\greekABC}$
\item[bold italic sans] \verb|\mathsfbfit| \\
  $\mathsfbfit{\abc}$ $\mathsfbfit{\ABC}$ $\mathsfbfit{\nums}$ \\
  $\mathsfbfit{\greekabc}$ $\mathsfbfit{\greekABC}$
\item[typewriter] \verb|\mathtt| \\
  $\mathtt{\abcx}$ $\mathtt{\ABC}$ $\mathtt{\nums}$ \\
  $\mathtt{\greekABC}$
\item[caligraphic] \verb|\mathcal| \\
  $\mathcal{\ABC}$
\item[blackboard bold] \verb|\mathbb| \\
  $\mathbb{\abc}$ $\mathbb{\ABC}$
\end{description}
